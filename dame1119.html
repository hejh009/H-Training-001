import torch


# Sigmoid
def sigmoid(x):
    return 1 / (1 + torch.exp(-x))


def y_function(x1, x2):
    term1 = sigmoid(3 * sigmoid(x1 * x1 + 2 * x2 + 1))
    term2 = sigmoid(x1 * x1 + 2 * x2 + 1)
    return term1 + term2 + 1


# x1x2
x1 = torch.tensor(2., requires_grad=True)
x2 = torch.tensor(5., requires_grad=True)

# y_functionx1x2
y = y_function(x1, x2)

# y_functionx1x2
y.backward()
